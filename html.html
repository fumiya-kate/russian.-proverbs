  const proverbs = [

    { ru: "Терпение и труд всё перетрут.", ja: "忍耐と努力はすべてを乗り越える。" },

    { ru: "Без труда не вытащишь и рыбку из пруда.", ja: "努力なしに成果は得られない。" },

    { ru: "Семь раз отмерь, один раз отрежь.", ja: "七度測って一度切れ。" },

  const proverb = proverbs[index % proverbs.length];

  document.getElementById("translation").textContent = proverb.ja;
